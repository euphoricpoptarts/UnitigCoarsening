#IMPORTANT: Use the cmake flag -DCMAKE_CXX_COMPILER=/path/to/your/nvcc_wrapper
cmake_minimum_required(VERSION 3.13)
project(sgpar CXX)
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED True)

#change the first parameter to wherever your kokkos and kokkos-kernels are located
#add_subdirectory(../kokkos bin_kokkos/)
#add_subdirectory(../kokkos-kernels bin_kernels/)
find_package(KokkosKernels REQUIRED)
link_libraries(Kokkos::kokkos Kokkos::kokkoskernels)

add_executable(hec_spec sgpar.cpp)
add_executable(hec_spec_spgemm sgpar.cpp)
add_executable(hec_spec_hashmap sgpar.cpp)
add_executable(match_spec sgpar.cpp)
add_executable(mtmetis_spec sgpar.cpp)
add_executable(mis_spec sgpar.cpp)
add_executable(hec_fm sgpar.cpp)
add_executable(match_fm sgpar.cpp)
add_executable(mtmetis_fm sgpar.cpp)
add_executable(mis_fm sgpar.cpp)

#possible flags are FM, RADIX, HASHMAP, RECOARSEN, MTMETIS, EXPERIMENT, USE_GNU_PARALLELMODE, _KOKKOS
#add_compile_definitions(_KOKKOS EXPERIMENT USE_GNU_PARALLELMODE RADIX RECOARSEN)
target_compile_definitions(hec_spec PUBLIC _KOKKOS EXPERIMENT USE_GNU_PARALLELMODE HEC RADIX)
target_compile_definitions(hec_spec_spgemm PUBLIC _KOKKOS EXPERIMENT USE_GNU_PARALLELMODE HEC SPGEMM)
target_compile_definitions(hec_spec_hashmap PUBLIC _KOKKOS EXPERIMENT USE_GNU_PARALLELMODE HEC HASHMAP)
target_compile_definitions(match_spec PUBLIC _KOKKOS EXPERIMENT USE_GNU_PARALLELMODE PUREMATCH RADIX)
target_compile_definitions(mtmetis_spec PUBLIC _KOKKOS EXPERIMENT USE_GNU_PARALLELMODE MTMETIS RADIX)
target_compile_definitions(mis_spec PUBLIC _KOKKOS EXPERIMENT USE_GNU_PARALLELMODE MIS RADIX)
target_compile_definitions(hec_fm PUBLIC _KOKKOS EXPERIMENT USE_GNU_PARALLELMODE HEC RADIX FM)
target_compile_definitions(match_fm PUBLIC _KOKKOS EXPERIMENT USE_GNU_PARALLELMODE PUREMATCH RADIX FM)
target_compile_definitions(mtmetis_fm PUBLIC _KOKKOS EXPERIMENT USE_GNU_PARALLELMODE MTMETIS RADIX FM)
target_compile_definitions(mis_fm PUBLIC _KOKKOS EXPERIMENT USE_GNU_PARALLELMODE MIS RADIX FM)
add_compile_options(-O3)
